
CREATE DATABASE todoapp;

CREATE TABLE project(
	 id int NOT NULL IDENTITY(1,1) PRIMARY KEY, 
	 name  varchar(50) NOT NULL,
	 description varchar(50),
	 createdAt  datetime,
	 updateAt  datetime

)

CREATE TABLE task(
	 id int NOT NULL IDENTITY(1,1) PRIMARY KEY,
	 idProject int,
	 name  varchar(50) NOT NULL,
	 description varchar(50),
	 completed  bit,
	 notes  varchar(50),
     deadline  date,
	 createdAt  datetime,
	 updateAt  datetime,
	 CONSTRAINT FK_idProject FOREIGN KEY (idProject)
    REFERENCES project(id)

)
